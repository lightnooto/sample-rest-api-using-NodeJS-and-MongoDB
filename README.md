# Rest API using NodeJS

Model : 
- Book
- Label
- Role
- User

Routes :
- Book :
  - /book = POST
  - /book = GET
  - /book/:bookId = GET
  - /book/:bookId = PUT
  - /book/:bookId = DELETE
  - /book/img/:bookId = GET
- Label :
  - /label = POST
  - /label = GET
  - /label/:labelId = GET
  - /findlabel = GET
  - /label/:labelId = PUT
  - /label/:labelId = DELETE  
- Role :
  - /role = POST
  - /role = GET
  - /role/:roleId = GET
  - /role/:roleId = PUT
  - /role/:roleId = DELETE  
 - User :
  - /user = POST
  - /user = GET
  - /user/:userId = GET
  - /user/:userId = PUT
  - /user/:userId = DELETE  
  - /login = POST
